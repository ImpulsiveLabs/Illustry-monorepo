  tagline: <strong>Something went wrong.</strong> We couldn’t find that page.<br>Check the URL or try using the search bar.
  actions:

      icon: right-arrow
      link: /
      variant: primary

<script is:inline>
  (function () {
    var base = '/Illustry-monorepo';
    var pathname = window.location.pathname;
    var englishPrefix = base + '/en';

    if (pathname === englishPrefix || pathname === englishPrefix + '/') {
      window.location.replace(base + '/');
      return;
    }

    if (pathname.indexOf(englishPrefix + '/') === 0) {
      window.location.replace(base + pathname.slice(englishPrefix.length));
    }
  })();
</script>
